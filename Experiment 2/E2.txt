class Solution {
public:
    double myPow(double x, int n) {
        double a=1;
        long long p=n;
        if(n<0){
            p=-p;
            x=1/x;
        }
        while(p>0){
            if(p&1){
                a*=x;
            }
            x=x*x;
            p>>=1;
        }
        
        return a;
        }
};
